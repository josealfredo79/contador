📋 src/lib.rs para copiar 
#![no_std]
use soroban_sdk::{contract, contractimpl, symbol_short, Env, Symbol};

const COUNTER: Symbol = symbol_short!("COUNTER");

#[contract]
pub struct ContadorContract;

#[contractimpl]
impl ContadorContract {
    pub fn increment(env: Env) -> u32 {
        let mut count: u32 = env.storage()
            .instance()
            .get(&COUNTER)
            .unwrap_or(0);
        count += 1;
        env.storage().instance().set(&COUNTER, &count);
        count
    }

    pub fn decrement(env: Env) -> u32 {
        let count: u32 = env.storage()
            .instance()
            .get(&COUNTER)
            .unwrap_or(0);
        let new_count = count.saturating_sub(1);
        env.storage().instance().set(&COUNTER, &new_count);
        new_count
    }

    pub fn get(env: Env) -> u32 {
        env.storage()
            .instance()
            .get(&COUNTER)
            .unwrap_or(0)
    }

    pub fn reset(env: Env) {
        env.storage().instance().set(&COUNTER, &0u32);
    }
}

#[cfg(test)]
mod test {
    use super::*;
    use soroban_sdk::Env;

    #[test]
    fn test_increment() {
        let env = Env::default();
        let contract_id = env.register(ContadorContract, ());
        let client = ContadorContractClient::new(&env, &contract_id);

        assert_eq!(client.increment(), 1);
        assert_eq!(client.increment(), 2);
        assert_eq!(client.get(), 2);
    }

    #[test]
    fn test_decrement() {
        let env = Env::default();
        let contract_id = env.register(ContadorContract, ());
        let client = ContadorContractClient::new(&env, &contract_id);

        client.increment();
        client.increment();
        client.increment();
        assert_eq!(client.get(), 3);
        assert_eq!(client.decrement(), 2);
        assert_eq!(client.decrement(), 0);
    }
}

Explicación Línea por Línea

1 #![no_std]

Indica que el código NO usa la librería estándar de Rust. Obligatorio para Soroban - reduce el tamaño del binario WASM.

2 use soroban_sdk::{...}

Importa las herramientas del SDK de Soroban:
• contract - Macro para definir un contrato
• contractimpl - Macro para implementar funciones
• symbol_short - Macro para crear claves de storage optimizadas
• Env - El entorno de ejecución de Soroban
• Symbol - Tipo para claves de almacenamiento

3 const COUNTER: Symbol

Define la clave para guardar el valor en el storage. symbol_short! crea un identificador de máximo 10 caracteres (optimizado para storage).

4 #[contract]

Macro que marca la estructura como un smart contract. pub struct hace que sea accesible públicamente.

5 #[contractimpl]

Macro que implementa las funciones del contrato. Todo dentro de este bloque será expuesto como funciones del smart contract.

6 pub fn increment(env: Env) -> u32

Define la función increment:
• pub fn - Función pública (accesible desde fuera)
• env: Env - Parámetro obligatorio de Soroban
• -> u32 - Retorna un número de 32 bits sin signo

7 env.storage().instance().get(&COUNTER)

Lee el valor actual del contador desde el storage:
• env.storage() - Accede al almacenamiento del contrato
• .instance() - Storage específico para esta instancia
• .get(&COUNTER) - Recupera el valor con la clave COUNTER
• .unwrap_or(0) - Si no existe, retorna 0 (valor inicial)

8 count += 1

Incrementa el contador en 1. La variable debe ser mut (mutable).

9 env.storage().instance().set(&COUNTER, &count)

Guarda el nuevo valor en el storage:
• .set(&COUNTER, &count) - Guarda el valor con la clave COUNTER

10 saturating_sub(1)

En la función decrement, evita el underflow:
• Si count es 0, retorna 0 (no valores negativos)
• Previene errores de wraparound en enteros

🔌 API Node.js

1. Crear proyecto API

mkdir api-contador && cd api-contador npm init -y npm install @stellar/stellar-sdk express dotenv

2. Variables de entorno (.env)

CONTRACT_ID=tu_contract_id PUBLIC_KEY=tu_public_key SECRET_KEY=tu_secret_key

3. Código del servidor (index.js)

const express = require('express'); const StellarSdk = require('@stellar/stellar-sdk'); require('dotenv').config(); const app = express(); app.use(express.json()); const server = new StellarSdk.SorobanRpc.Server('https://soroban-testnet.stellar.org'); const contract = new StellarSdk.Contract(process.env.CONTRACT_ID); app.get('/contador', async (req, res) => { try { const account = await server.getAccount(process.env.PUBLIC_KEY); const tx = new StellarSdk.TransactionBuilder(account, { fee: StellarSdk.BASE_FEE, networkPassphrase: StellarSdk.Networks.TESTNET }) .addOperation(contract.call('get')) .setTimeout(30) .build(); const sim = await server.simulateTransaction(tx); const value = StellarSdk.scValToNative(sim.result?.retval); res.json({ count: value }); } catch (error) { res.status(500).json({ error: error.message }); } }); app.post('/contador/increment', async (req, res) => { // Similar a GET pero con contract.call('increment') }); app.listen(3000, () => console.log('API corriendo en puerto 3000'));

🔧 Solución de Problemas

Error: target not found

rustup target add wasm32v1-none

Error: wasm32-unknown-unknown unsupported

Usar wasm32v1-none en lugar de wasm32-unknown-unknown

rustup target add wasm32v1-none

Error: insufficient funds

Solicita XLM de prueba en https://laboratory.stellar.org

Error: contract not found

Verifica que el CONTRACT_ID sea correcto y esté desplegado en testnet.

Compilación lenta

Usa cargo build --release solo cuando sea necesario.

Tests fallan

Revisa que uses env.register(ContadorContract, ()) en lugar de register_contract
